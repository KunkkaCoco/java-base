public class Triangle {
	
	public int function(int n){
		int a=0;
		if(n>0){
			a= n*function(n-1);
		}else {
			a = 1;
		}
		
		return a;
	}

	/**
	 * @param args
	 */
	public static void main(String[] args) {
		Triangle tri = new Triangle();
		int i=0;
		i = tri.function(5);
		System.out.println(i);

	}

}
